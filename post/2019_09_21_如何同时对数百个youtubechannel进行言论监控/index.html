【项目实战】入门教学，如何发送你的第一段异步请求。

October 24, 2018
Page content
import requests

tasks = ['http://www.baidu.com']*20

for url in tasks:
    requests.get(url)
import aiohttp
import asyncio


async def fetch(url):
    async with aiohttp.ClientSession() as session:
        await session.get(url)

async def run_all():
    await asyncio.gather(*[fetch(url) for url in tasks ])

asyncio.run(run_all())
