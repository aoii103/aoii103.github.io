import string
import random
from execjs import eval as jseval

def random_key(length=20):
    return "".join(
        (
            random.choice(
                random.choice(
                    (
                        string.ascii_uppercase,
                        string.ascii_lowercase,
                    )
                )
            )
            for i in range(1, length)
        )
    )
